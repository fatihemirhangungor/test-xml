    {
        using (var stream = new FileStream(sFileXlsEft, FileMode.Open, FileAccess.Read))
        {
            IWorkbook workbook = new XSSFWorkbook(stream);
            ISheet sheet = workbook.GetSheetAt(0);
            DataTable dtEft = new DataTable();
            
            bool firstRow = true;
            foreach (IRow row in sheet)
            {
                if (firstRow)
                {
                    foreach (ICell cell in row.Cells)
                    {
                        dtEft.Columns.Add(cell.StringCellValue);
                    }
                    firstRow = false;
                }
                else
                {
                    var dataRow = dtEft.NewRow();
                    int i = 0;
                    foreach (ICell cell in row)
                    {
                        dataRow[i] = cell.StringCellValue;
                        i++;
                    }
                    dtEft.Rows.Add(dataRow);
                }
            }
        }
    }
